def huella_ambiental
    energia = 0.0
    carbono = 0.0
    huella = 0.0
    if vct < 670 then
        energia = 1.0
    elsif vct <= 830 then
        energia = 2.0
    else
        energia = 3.0
    end
    if gei < 800 then
        carbono = 1.0
    elsif terrenos <= 1200 then
        carbono = 2.0
    else
        carbono = 3.0
    end
    huella = (energia + carbono)/2
    return huella
    
end

def to_plato_s
    aaa = ""
    aaa +="Plato: #{@nombre} \n"
    @lista_alimentos.each do |i|
        aaa +="Valor energético: #{i.val_ener} \n"
    end
    return aaa
end
