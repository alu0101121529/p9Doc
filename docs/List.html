def each

    while(node!=nil) do
        yield node.value
        node = node.next
    end
end

def extract_tail
    if @tail!=nil then
        temp = @tail
        @tail = @tail.prev
        if @tail!=nil then
            @tail.next = nil
        end
        return temp
    end
end

def insert_head(value)
    if (head==nil) then
        @head=Node.new(value,nil,nil)
        @tail=@head
    else
        temp=Node.new(value,@head,nil)
        @head.prev=temp
        @head=temp
    end
end
